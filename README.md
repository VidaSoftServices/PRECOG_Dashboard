# install
Initialize React App
npm create vite@latest precog-dashboard -- --template react
cd precog-dashboard
npm install(https://swc.rs/) for Fast Refresh

npm run dev

cd C:\Docker\Dashboard\PRECOG\PRECOG\dashboard
docker-compose build
